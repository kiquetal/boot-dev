# Modify the current kiali service to use Nodeport instead of ClusterIP
# The service uses 2 ports (20001 for UI and 9090 for metrics)
kubectl patch svc kiali -n istio-system -p '{"spec": {"type": "NodePort", "ports": [{"port": 20001, "nodePort": 30001, "protocol": "TCP", "targetPort": 20001}, {"port": 9090, "nodePort": 30090, "protocol": "TCP", "targetPort": 9090}]}}'
